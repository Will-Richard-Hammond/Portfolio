Keybinds

"W" / "Up Arrow" = "Move forward"
"S" / "Down Arrow" = "Move backward"
"A" / "Left Arrow" = "Strafe left"
"D" / "Right Arrow" = "Strafe right"
"Mouse Move" = "Turn camera horizontally"
"Left Mouse Button" = "Fire weapon"
"B" = "Toggle birdseye view"
"M" = "Toggle drone movement on/off"
"C" = "Toggle player wall collision on/off"
"Escape" = "Return to main menu"

HUD

"Lives" = "Remaining player lives"
"Weapon" = "Current weapon damage level"
"Drones" = "Drone status, [ ] active and [X] disabled"
"Drone Move" = "Shows whether drone movement is enabled"
"Wall Collide" = "Shows whether player wall collision is enabled"
"EMP" = "Remaining temporary drone disable time"
